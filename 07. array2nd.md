### 행렬 덧셈
```py
N, M = map(int, input().split())
sub = []
sub2 = []
total = []
for _ in range(N):
  line = list(map(int, input().split()))
  sub.append(line)
for _ in range(N):
  line2 = list(map(int, input().split()))
  sub2.append(line2)
for _ in range(N):
  for __ in range(M):
    total.append(sub[_][__] + sub2[_][__])
for __ in range(M):
  print(*total[(__)*M:(__ + 1)*M], sep=' ')
```
